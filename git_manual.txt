# 1. Git 초기화 (이미 되어 있다면 생략 가능)
git init

# 2. 원격 저장소 연결 (이미 연결되어 있다면 set-url로 변경 가능)
git remote add origin https://github.com/mr-newbe/kisia_ai_security_hackerton.git

# 3. 로컬 변경 사항 커밋
git add .
git commit -m "작업 내용 반영"

# 4. 원격 저장소의 최신 커밋을 받아와 병합 (충돌 최소화)
git pull origin main --rebase

# 5. 병합된 내용을 푸시
git push origin main
